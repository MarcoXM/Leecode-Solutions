# 1573. Number of Ways to Split a String

```python
class Solution:
    def numWays(self, s: str) -> int:
        ones = s.count("1")
        m = 10**9+7
        if ones == 0 :
            return (len(s) - 1) * (len(s) - 2)//2% m
        
        elif  ones%3:
            return 0


        count = p1 = p2 = 0
        ones = ones//3
        for c in s:
            if c =="1":
                count += 1
            if count == ones:
                p1 += 1

            elif count == 2 * ones:
                p2 += 1

        return p1 * p2 %m
                    
    
```
