# The rand7() API is already defined for you.
# def rand7():
# @return a random integer in the range 1 to 7

class Solution:
    def rand10(self):
        """
        :rtype: int
        """
        ran = 40
        while ran>=40:
            ran = ((rand7() - 1) * 7) + rand7() - 1
            
        return ran%10 + 1
