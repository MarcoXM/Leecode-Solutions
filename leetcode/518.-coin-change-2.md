int dp[5010];
class Solution {
public:
    int change(int amount, vector<int>& coins) {
        //还是要找到关系， 假设目标是5块，那么dp[5] += dp[i] for i in range(coins)
        
        memset(dp,0 , sizeof(dp));
        dp[0] = 1;
        for (int c:coins){
             for (int i = 1; i <= amount; i++) {
                if (i - c < 0) continue;
                dp[i] += dp[i-c];
            }
        }
        return dp[amount];
    }
};
